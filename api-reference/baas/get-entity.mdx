---
title: "Get Entity"
description: "Retrieve details of an existing BaaS entity"
api: "GET https://api-sandbox.y.uno/v1/baas/entities/{entity_id}"
---

## Path parameters

<ParamField path="entity_id" type="string" required>
  The unique identifier of the entity obtained from the Create Entity endpoint.
</ParamField>

## Response

<ResponseField name="id" type="string">
  Entity identifier.
</ResponseField>

<ResponseField name="account_id" type="string">
  Merchant's Yuno account identifier.
</ResponseField>

<ResponseField name="merchant_entity_id" type="string">
  Merchant's own identifier for reconciliation.
</ResponseField>

<ResponseField name="national_entity" type="string">
  `INDIVIDUAL` or `ENTITY`.
</ResponseField>

<ResponseField name="phone" type="object">
  Phone details with `country_code` and `number`.
</ResponseField>

<ResponseField name="address" type="object">
  Address with `address_line_1`, `city`, `state`, `zip_code`, `country`.
</ResponseField>

<ResponseField name="entity_detail" type="object">
  Individual or business details.
</ResponseField>

<ResponseField name="created_at" type="string">
  ISO 8601 timestamp.
</ResponseField>

<ResponseField name="updated_at" type="string">
  ISO 8601 timestamp.
</ResponseField>
